57 {
58  struct nk_sfml_device* dev = &sfml.ogl;
59  glGenTextures(1, &dev->font_tex);
60  glBindTexture(GL_TEXTURE_2D, dev->font_tex);
61  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
62  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
63  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, (GLsizei)width, (GLsizei)height, 0,
64  GL_RGBA, GL_UNSIGNED_BYTE, image);

67 NK_API void
68 nk_sfml_render(enum nk_anti_aliasing AA)
69 {
70  /* setup global state */
71  struct nk_sfml_device* dev = &sfml.ogl;
72 
73  int window_width = sfml.window->getSize().x;
74  int window_height = sfml.window->getSize().y;
75 
76  glPushAttrib(GL_ENABLE_BIT | GL_COLOR_BUFFER_BIT | GL_TRANSFORM_BIT);
77  glDisable(GL_CULL_FACE);
78  glDisable(GL_DEPTH_TEST);
79  glEnable(GL_SCISSOR_TEST);
80  glEnable(GL_BLEND);
81  glEnable(GL_TEXTURE_2D);
82  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
83 
84  glViewport(0, 0, (GLsizei)window_width, (GLsizei)window_height);
85  glMatrixMode(GL_TEXTURE);
86  glPushMatrix();
87  glLoadIdentity();
88  glMatrixMode(GL_PROJECTION);
89  glPushMatrix();
90  glLoadIdentity();
91  glOrtho(0.0f, window_width, window_height, 0.0f, -1.0f, 1.0f);
92  glMatrixMode(GL_MODELVIEW);
93  glPushMatrix();
94  glLoadIdentity();
95 
96  glEnableClientState(GL_VERTEX_ARRAY);
97  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
98  glEnableClientState(GL_COLOR_ARRAY);
99  {
100  GLsizei vs = sizeof(struct nk_sfml_vertex);
101  size_t vp = NK_OFFSETOF(struct nk_sfml_vertex, position);
102  size_t vt = NK_OFFSETOF(struct nk_sfml_vertex, uv);
103  size_t vc = NK_OFFSETOF(struct nk_sfml_vertex, col);
104 
105  /* convert from command queue into draw list and draw to screen */
106  const struct nk_draw_command* cmd;
107  const nk_draw_index* offset = NULL;
108  struct nk_buffer vbuf, ebuf;
109 
110  /* fill converting configuration */
111  struct nk_convert_config config;
112  static const struct nk_draw_vertex_layout_element vertex_layout[] = {
113  {NK_VERTEX_POSITION, NK_FORMAT_FLOAT, NK_OFFSETOF(struct nk_sfml_vertex, position)},
114  {NK_VERTEX_TEXCOORD, NK_FORMAT_FLOAT, NK_OFFSETOF(struct nk_sfml_vertex, uv)},
115  {NK_VERTEX_COLOR, NK_FORMAT_R8G8B8A8, NK_OFFSETOF(struct nk_sfml_vertex, col)},
116  {NK_VERTEX_LAYOUT_END}
117  };
118  NK_MEMSET(&config, 0, sizeof(config));
119  config.vertex_layout = vertex_layout;
120  config.vertex_size = sizeof(struct nk_sfml_vertex);
121  config.vertex_alignment = NK_ALIGNOF(struct nk_sfml_vertex);
122  config.null = dev->null;
123  config.circle_segment_count = 22;
124  config.curve_segment_count = 22;
125  config.arc_segment_count = 22;
126  config.global_alpha = 1.0f;
127  config.shape_AA = AA;
128  config.line_AA = AA;
129 
130  /* convert shapes into vertices */
131  nk_buffer_init_default(&vbuf);
132  nk_buffer_init_default(&ebuf);
133  nk_convert(&sfml.ctx, &dev->cmds, &vbuf, &ebuf, &config);
134 
135  /* setup vertex buffer pointer */
136  const void* vertices = nk_buffer_memory_const(&vbuf);
137  glVertexPointer(2, GL_FLOAT, vs, (const void*)((const nk_byte*)vertices + vp));
138  glTexCoordPointer(2, GL_FLOAT, vs, (const void*)((const nk_byte*)vertices + vt));
139  glColorPointer(4, GL_UNSIGNED_BYTE, vs, (const void*)((const nk_byte*)vertices + vc));
140 
141  /* iterate over and execute each draw command */
142  offset = (const nk_draw_index*)nk_buffer_memory_const(&ebuf);
143  nk_draw_foreach(cmd, &sfml.ctx, &dev->cmds)
144  {
145  if(!cmd->elem_count) continue;
146  glBindTexture(GL_TEXTURE_2D, (GLuint)cmd->texture.id);
147  glScissor(
148  (GLint)(cmd->clip_rect.x),
149  (GLint)((window_height - (GLint)(cmd->clip_rect.y + cmd->clip_rect.h))),
150  (GLint)(cmd->clip_rect.w),
151  (GLint)(cmd->clip_rect.h));
152  glDrawElements(GL_TRIANGLES, (GLsizei)cmd->elem_count, GL_UNSIGNED_SHORT, offset);
153  offset += cmd->elem_count;
154  }
155  nk_clear(&sfml.ctx);
156  nk_buffer_free(&vbuf);
157  nk_buffer_free(&ebuf);
158  }
159 
160  /* default OpenGL state */
161  glDisableClientState(GL_VERTEX_ARRAY);
162  glDisableClientState(GL_TEXTURE_COORD_ARRAY);
163  glDisableClientState(GL_COLOR_ARRAY);
164 
165  glDisable(GL_CULL_FACE);
166  glDisable(GL_DEPTH_TEST);
167  glDisable(GL_SCISSOR_TEST);
168  glDisable(GL_BLEND);
169  glDisable(GL_TEXTURE_2D);
170 
171  glBindTexture(GL_TEXTURE_2D, 0);
172  glMatrixMode(GL_TEXTURE);
173  glPopMatrix();
174  glMatrixMode(GL_MODELVIEW);
175  glPopMatrix();
176  glMatrixMode(GL_PROJECTION);
177  glPopMatrix();
178  glPopAttrib();
179 }

246 NK_API int
247 nk_sfml_handle_event(sf::Event* evt)
248 {
249  struct nk_context* ctx = &sfml.ctx;
250  /* optional grabbing behavior */
251  if(ctx->input.mouse.grab)
252  ctx->input.mouse.grab = 0;
253  else if(ctx->input.mouse.ungrab) {
254  int x = (int)ctx->input.mouse.prev.x;
255  int y = (int)ctx->input.mouse.prev.y;
256  sf::Mouse::setPosition(sf::Vector2i(x, y), *sfml.window);
257  ctx->input.mouse.ungrab = 0;
258  }
259  if(evt->type == sf::Event::KeyReleased || evt->type == sf::Event::KeyPressed)
260  {
261  int down = evt->type == sf::Event::KeyPressed;
262  sf::Keyboard::Key key = evt->key.code;
263  if(key == sf::Keyboard::RShift || key == sf::Keyboard::LShift)
264  nk_input_key(ctx, NK_KEY_SHIFT, down);
265  else if(key == sf::Keyboard::Delete)
266  nk_input_key(ctx, NK_KEY_DEL, down);
267  else if(key == sf::Keyboard::Return)
268  nk_input_key(ctx, NK_KEY_ENTER, down);
269  else if(key == sf::Keyboard::Tab)
270  nk_input_key(ctx, NK_KEY_TAB, down);
271  else if(key == sf::Keyboard::BackSpace)
272  nk_input_key(ctx, NK_KEY_BACKSPACE, down);
273  else if(key == sf::Keyboard::Home) {
274  nk_input_key(ctx, NK_KEY_TEXT_START, down);
275  nk_input_key(ctx, NK_KEY_SCROLL_START, down);
276  } else if(key == sf::Keyboard::End) {
277  nk_input_key(ctx, NK_KEY_TEXT_END, down);
278  nk_input_key(ctx, NK_KEY_SCROLL_END, down);
279  } else if(key == sf::Keyboard::PageDown)
280  nk_input_key(ctx, NK_KEY_SCROLL_DOWN, down);
281  else if(key == sf::Keyboard::PageUp)
282  nk_input_key(ctx, NK_KEY_SCROLL_DOWN, down);
283  else if(key == sf::Keyboard::Z)
284  nk_input_key(ctx, NK_KEY_TEXT_UNDO, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
285  else if(key == sf::Keyboard::R)
286  nk_input_key(ctx, NK_KEY_TEXT_REDO, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
287  else if(key == sf::Keyboard::C)
288  nk_input_key(ctx, NK_KEY_COPY, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
289  else if(key == sf::Keyboard::V)
290  nk_input_key(ctx, NK_KEY_PASTE, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
291  else if(key == sf::Keyboard::X)
292  nk_input_key(ctx, NK_KEY_CUT, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
293  else if(key == sf::Keyboard::B)
294  nk_input_key(ctx, NK_KEY_TEXT_LINE_START, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
295  else if(key == sf::Keyboard::E)
296  nk_input_key(ctx, NK_KEY_TEXT_LINE_END, down && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl));
297  else if(key == sf::Keyboard::Up)
298  nk_input_key(ctx, NK_KEY_UP, down);
299  else if(key == sf::Keyboard::Down)
300  nk_input_key(ctx, NK_KEY_DOWN, down);
301  else if(key == sf::Keyboard::Left) {
302  if(sf::Keyboard::isKeyPressed(sf::Keyboard::LControl))
303  nk_input_key(ctx, NK_KEY_TEXT_WORD_LEFT, down);
304  else nk_input_key(ctx, NK_KEY_LEFT, down);
305  } else if(key == sf::Keyboard::Right) {
306  if(sf::Keyboard::isKeyPressed(sf::Keyboard::LControl))
307  nk_input_key(ctx, NK_KEY_TEXT_WORD_RIGHT, down);
308  else nk_input_key(ctx, NK_KEY_RIGHT, down);
309  } else return 0;
310  return 1;
311  } else if(evt->type == sf::Event::MouseButtonPressed || evt->type == sf::Event::MouseButtonReleased) {
312  int down = evt->type == sf::Event::MouseButtonPressed;
313  const int x = evt->mouseButton.x, y = evt->mouseButton.y;
314  if(evt->mouseButton.button == sf::Mouse::Left)
315  nk_input_button(ctx, NK_BUTTON_LEFT, x, y, down);
316  if(evt->mouseButton.button == sf::Mouse::Middle)
317  nk_input_button(ctx, NK_BUTTON_MIDDLE, x, y, down);
318  if(evt->mouseButton.button == sf::Mouse::Right)
319  nk_input_button(ctx, NK_BUTTON_RIGHT, x, y, down);
320  else return 0;
321  return 1;
322  } else if(evt->type == sf::Event::MouseMoved) {
323  nk_input_motion(ctx, evt->mouseMove.x, evt->mouseMove.y);
324  return 1;
325  } else if(evt->type == sf::Event::TouchBegan || evt->type == sf::Event::TouchEnded) {
326  int down = evt->type == sf::Event::TouchBegan;
327  const int x = evt->touch.x, y = evt->touch.y;
328  ctx->input.mouse.pos.x = x;
329  ctx->input.mouse.pos.y = y;
330  nk_input_button(ctx, NK_BUTTON_LEFT, x, y, down);
331  return 1;
332  } else if(evt->type == sf::Event::TouchMoved) {
333  if(ctx->input.mouse.grabbed) {
334  int x = (int)ctx->input.mouse.prev.x;
335  int y = (int)ctx->input.mouse.prev.y;
336  nk_input_motion(ctx, x + evt->touch.x, y + evt->touch.y);
337  } else nk_input_motion(ctx, evt->touch.x, evt->touch.y);
338  return 1;
339  } else if(evt->type == sf::Event::TextEntered) {
340  nk_input_unicode(ctx, evt->text.unicode);
341  return 1;
342  } else if(evt->type == sf::Event::MouseWheelScrolled) {
343  nk_input_scroll(ctx, nk_vec2(0,evt->mouseWheelScroll.delta));
344  return 1;
345  }
346  return 0;
347 }
